## <a name="example"></a><span data-ttu-id="63aa6-109">Beispiel</span><span class="sxs-lookup"><span data-stu-id="63aa6-109">Example</span></span>

{  
    static void Main()  
    {  
        int[] array = new int[] { 1, 2, 3, 4, 5 };
