## <a name="example"></a><span data-ttu-id="c351b-107">Beispiel</span><span class="sxs-lookup"><span data-stu-id="c351b-107">Example</span></span>

{  
    // Both of these declarations trigger CS0825
    private var genreName;
    private var bookTitles = new List<string>();

    }  
}
