## <a name="example"></a><span data-ttu-id="fbae9-109">Beispiel</span><span class="sxs-lookup"><span data-stu-id="fbae9-109">Example</span></span>

{  
    public static S ExtMethod(this S s)  
    {  
        return s;  
    }

{  
}
